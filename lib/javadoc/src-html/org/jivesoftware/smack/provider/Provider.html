018package org.jivesoftware.smack.provider;
019
020import org.jivesoftware.smack.packet.Element;
021import org.jivesoftware.smack.util.ParserUtils;
022
023import org.xmlpull.v1.XmlPullParser;
024
025/**
026 * Smack provider are the parsers used to deserialize raw XMPP into the according Java {@link Element}s.
027 * <p>
028 * At any time when {@link #parse(XmlPullParser, int)} is invoked any type of exception can be thrown. If the parsed
029 * element does not follow the specification, for example by putting a string where only integers are allowed, then a
030 * {@link org.jivesoftware.smack.SmackException} should be thrown.
031 * </p>
032 *
033 * @author Florian Schmaus
034 * @param <E> the type of the resulting element.
035 */
036public abstract class Provider<E extends Element> {
037
038    public final E parse(XmlPullParser parser) throws Exception {
039        // XPP3 calling convention assert: Parser should be at start tag
040        ParserUtils.assertAtStartTag(parser);
041
042        final int initialDepth = parser.getDepth();
043        E e = parse(parser, initialDepth);
044
045        // XPP3 calling convention assert: Parser should be at end tag of the consumed/parsed element
046        ParserUtils.forwardToEndTagOfDepth(parser, initialDepth);
047        return e;
048    }
049
050    public abstract E parse(XmlPullParser parser, int initialDepth) throws Exception;
051}
